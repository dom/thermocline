   ```python
   # Replacement for _verify_brine — real ed25519 verification via thermocline.identity
   def _verify_brine(body: dict, sig_block: dict, keyring_service: str | None = None) -> None:
       """Verify an ed25519 (brine) dispatch signature using thermocline.identity.Verifier."""
       from thermocline.identity import Signature
       from thermocline.schemes import KeyScheme
       from .forge_identity import get_verifier

       node_id = sig_block.get("node_id") or sig_block.get("signer_identity") or ""
       sig_hex = sig_block.get("sig") or sig_block.get("bytes_hex") or ""
       if not sig_hex:
           raise EnvelopeError(
               "SIGNATURE_INVALID",
               "dispatch_signature.sig is empty for key_scheme=brine",
               http_status=401,
           )
       sig = Signature(
           scheme=KeyScheme.BRINE,
           bytes_=bytes.fromhex(sig_hex),
           signer_identity=node_id,
       )
       verifier = get_verifier(keyring_service)
       # Per Phase 1 BL-01: verifier looks up signer's pubkey via register_public_key entries
       # in the same keyring service. The forge must have called register_public_key for the
       # signer at bootstrap (or via channel_new TOFU on the sovereign side, mirrored here).
       receipt = verifier.verify(envelope=body, signature=sig)
       if receipt is None:
           raise EnvelopeError(
               "SIGNATURE_INVALID",
               "dispatch_signature verification failed",
               http_status=401,
           )

   # Replacement for _sign_receipt — real ed25519 signing via BrineProvider + canonicalize
   def _sign_receipt(
       *,
       key_scheme: str,
       responder: str,
       envelope_id: str,
       result_id: str,
       timestamp: str,
       result_envelope_for_signing: dict | None = None,
       keyring_service: str | None = None,
   ) -> dict:
       """Build a receipt_signature block. Uses BrineProvider.sign over canonicalize() output."""
       base = {
           "key_scheme": key_scheme,
           "node_id": responder,
           "envelope_id": envelope_id,
           "result_id": result_id,
           "timestamp": timestamp,
       }
       if key_scheme == "none":
           base["sig"] = None
           return base
       if key_scheme != "brine":
           base["sig"] = None
           return base
       # Brine path: sign canonical-JSON of the entire result envelope (sans receipt_signature.sig).
       from .forge_identity import get_provider
       provider = get_provider(keyring_service)
       if result_envelope_for_signing is None:
           # Caller did not provide; sign over the receipt fields themselves as a fallback.
           signing_input_obj = base
       else:
           # Strip receipt_signature.sig before signing (cannot sign-over-self).
           import copy
           signing_input_obj = copy.deepcopy(result_envelope_for_signing)
           rs = signing_input_obj.get("receipt_signature", {})
           if isinstance(rs, dict):
               rs.pop("sig", None)
       sig = provider.sign(envelope=signing_input_obj, signer_identity=responder)
       base["sig"] = sig.bytes_.hex()
       return base
   ```

   ```python
   CONTEXT D-02: The string template is NORMATIVE — any deviation breaks FORGE-03.

   Privacy invariant T-03-11: this module NEVER reads or echoes inline content from

   `tier == 1 and "shadow" in block` BEFORE entering the description loop.
   """

   from typing import Any


   def filter_tier1_shadows(context: list[dict[str, Any]]) -> list[dict[str, Any]]:
       """Return only tier-1 blocks that carry a shadow payload."""
       return [
           b for b in context
           if isinstance(b, dict) and b.get("tier") == 1 and isinstance(b.get("shadow"), dict)
       ]


   def count_ignored_inline_blocks(context: list[dict[str, Any]]) -> int:
       """Return the number of non-shadow blocks (tier != 1 or missing shadow payload)."""
       return sum(
           1 for b in context
           if isinstance(b, dict) and not (b.get("tier") == 1 and isinstance(b.get("shadow"), dict))
       )


   def describe_one_shadow(shadow: dict[str, Any]) -> dict[str, Any]:
       """Return one ShadowDescription per CONTEXT D-02. Raises KeyError on malformed shadow."""
       shadow_id = shadow["shadow_id"]
       content_type = shadow["content_type"]
       relevance = shadow["relevance"]
       description = (

           f"with relevance {relevance}."
       )
       return {
           "shadow_id": shadow_id,
           "content_type": content_type,
           "relevance": relevance,
           "description": description,
       }

       """Iterate tier-1 shadows in context[] and return (descriptions, ignored_inline_count).

       Mixed-tier privacy invariant: inline (non-shadow) blocks contribute ONLY to

       shadows = filter_tier1_shadows(context)
       ignored = count_ignored_inline_blocks(context)
   ```
