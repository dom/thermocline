```python
"""RFC 8785 canonical JSON — the only path for signing input across the suite.

Pitfall 11: `json.dumps` is non-canonical (sort_keys default False, separators have spaces).
A signature computed over `json.dumps(envelope.model_dump())` does NOT verify across
implementations. Every signing/verifying path goes through `canonicalize`, period.
"""
from __future__ import annotations
from typing import Any
import rfc8785
from .errors import CanonicalizationError

__all__ = ["canonicalize", "CanonicalizationError"]

JsonScalar = str | int | float | bool | None
JsonValue = JsonScalar | list["JsonValue"] | dict[str, "JsonValue"]


def canonicalize(payload: Any) -> bytes:
    """Return RFC 8785 canonical-JSON bytes for `payload`.

    Args:
        payload: A value produced by Pydantic `model.model_dump(mode="json")` —
            recursively `dict[str, ...]`, `list[...]`, `str`, `int`, `float`,
            `bool`, or `None`. The `mode="json"` discipline ensures that
            `Sensitive[bytes]` has already been base64-encoded to `str`.

    Returns:
        Canonical-JSON bytes per RFC 8785. The same logical payload always
        produces the same bytes regardless of source-language map ordering.

    Raises:
        CanonicalizationError: on inputs containing non-JSON-serializable types
            (sets, datetimes, custom objects, raw `bytes`) or on internal
            `rfc8785` failure. This raises early so signing paths cannot
            silently fall back to `json.dumps`.
    """
    try:
        return rfc8785.dumps(payload)
    except (TypeError, ValueError) as exc:
        raise CanonicalizationError(
            f"payload is not canonical-JSON-serializable: {exc}",
            code="CANONICALIZATION_FAILED",
        ) from exc
```

```python
from __future__ import annotations
import json
from hypothesis import given, settings, strategies as st, HealthCheck
from thermocline.canonical import canonicalize
from thermocline.envelope import Task, ContentBlock
from thermocline.sensitive import Sensitive

# Strategy for arbitrary JSON-compatible scalars and recursive structures.
json_scalars = st.one_of(
    st.none(),
    st.booleans(),
    st.integers(min_value=-(10**12), max_value=10**12),
    st.floats(allow_nan=False, allow_infinity=False, width=64),
    st.text(min_size=0, max_size=64),
)
json_values = st.recursive(
    json_scalars,
    lambda children: st.one_of(
        st.lists(children, max_size=5),
        st.dictionaries(st.text(min_size=1, max_size=16), children, max_size=5),
    ),
    max_leaves=20,
)
json_dicts = st.dictionaries(st.text(min_size=1, max_size=16), json_values, min_size=0, max_size=8)


@given(payload=json_dicts)
@settings(max_examples=200, suppress_health_check=[HealthCheck.too_slow])
def test_canonicalize_idempotent_under_json_roundtrip(payload):
    """canonicalize(d) == canonicalize(d after json roundtrip) for all JSON-shaped dicts."""
    rebuilt = json.loads(json.dumps(payload))
    assert canonicalize(payload) == canonicalize(rebuilt)


@given(payload=json_dicts.filter(lambda d: len(d) > 0))
@settings(max_examples=100)
def test_canonicalize_detects_value_mutation(payload):
    """A single-leaf mutation must change the canonical bytes."""
    original = canonicalize(payload)
    # Pick the first key, mutate its value type-appropriately.
    key = next(iter(payload))
    val = payload[key]
    mutated = dict(payload)
    if isinstance(val, bool):
        mutated[key] = not val
    elif isinstance(val, int):
        mutated[key] = val + 1
    elif isinstance(val, float):
        mutated[key] = val + 1.0
    elif isinstance(val, str):
        mutated[key] = val + "x"
    elif val is None:
        mutated[key] = False
    elif isinstance(val, list):
        mutated[key] = list(val) + [None]
    elif isinstance(val, dict):
        mutated[key] = dict(val, __synthetic__=None)
    else:
        return  # nothing reasonable to mutate; skip without failing
    assert canonicalize(mutated) != original


@given(keys=st.lists(st.text(min_size=1, max_size=8), min_size=2, max_size=6, unique=True))
def test_canonicalize_normalizes_key_order(keys):
    """Reordering top-level dict keys must NOT change canonical bytes."""
    forward = {k: i for i, k in enumerate(keys)}
    reversed_keys = list(reversed(keys))
    backward = {k: forward[k] for k in reversed_keys}
    assert canonicalize(forward) == canonicalize(backward)


@given(items=st.lists(st.integers(), min_size=2, max_size=6, unique=True))
def test_canonicalize_preserves_list_order(items):
    """Reordering list elements MUST change canonical bytes (arrays are order-sensitive)."""
    forward_payload = {"items": items}
    reversed_payload = {"items": list(reversed(items))}
    assert canonicalize(forward_payload) != canonicalize(reversed_payload)


@given(content=st.binary(min_size=0, max_size=128))
@settings(max_examples=100)
def test_sensitive_wrapper_wire_transparent(content):
    """Sensitive[bytes] round-trips through canonicalize byte-for-byte deterministically.

    Demonstrates D-03: the Sensitive wrapper is a Python repr concern, NOT a wire-format change.
    """
    block = ContentBlock(content_id="cb-1", content_type="application/octet-stream", content=Sensitive(content))
    payload = block.model_dump(mode="json")
    once = canonicalize(payload)
    twice = canonicalize(payload)
    assert once == twice
    # Round-trip through JSON and re-canonicalize: bytes still equal.
    via_json = json.loads(block.model_dump_json())
    assert canonicalize(via_json) == once
```

**`src/thermocline/scripts/check_no_json_dumps.py`** (AST-based — substring matching falsely flags `# json.dumps` in comments and `not_json_dumps_helper` in identifiers):

```python
"""Pitfall 11 lint: forbid `json.dumps` in library code outside the allowlist.

Signing input MUST go through `thermocline.canonical.canonicalize` (RFC 8785).
`json.dumps` produces non-canonical output; signatures over its output do not
verify across implementations. This script walks `src/thermocline/` and flags
every Call node whose target is `json.dumps` or `json.dump`.

Allowlist (explicit, append-only):
- `src/thermocline/scripts/build_schemas.py` — committed schema artifacts use
  `json.dumps(..., indent=2, sort_keys=True)` for human-readable diffs. Schemas
  are NOT signing input; this is dev/CI tooling only.
- `src/thermocline/scripts/check_no_json_dumps.py` — this file (the lint can't
  exempt itself from itself in a substring check; AST-based check is safe).
"""
from __future__ import annotations
import ast
import sys
from pathlib import Path

ALLOWLIST = {
    "scripts/build_schemas.py",
    "scripts/check_no_json_dumps.py",
}

ROOT = Path(__file__).resolve().parents[2]  # src/thermocline/

class JsonDumpsVisitor(ast.NodeVisitor):
    def __init__(self) -> None:
        self.findings: list[tuple[int, str]] = []
    def visit_Call(self, node: ast.Call) -> None:
        # Match `json.dumps(...)` and `json.dump(...)` only.
        target = node.func
        if isinstance(target, ast.Attribute) and isinstance(target.value, ast.Name):
            if target.value.id == "json" and target.attr in {"dumps", "dump"}:
                self.findings.append((node.lineno, ast.unparse(node)))
        self.generic_visit(node)


def scan(path: Path) -> list[tuple[Path, int, str]]:
    findings: list[tuple[Path, int, str]] = []
    for py in path.rglob("*.py"):
        rel = py.relative_to(ROOT)
        if str(rel) in ALLOWLIST:
            continue
        if any(part.startswith(".") for part in rel.parts):
            continue
        try:
            tree = ast.parse(py.read_text(), filename=str(py))
        except SyntaxError:
            continue
        v = JsonDumpsVisitor()
        v.visit(tree)
        for lineno, src in v.findings:
            findings.append((rel, lineno, src))
    return findings


def main(argv: list[str] | None = None) -> int:
    findings = scan(ROOT)
    if findings:
        print("Pitfall 11 violation: json.dumps used in library code outside allowlist.\n", file=sys.stderr)
        for path, lineno, src in findings:
            print(f"  {path}:{lineno}: {src}", file=sys.stderr)
        print("\nUse `thermocline.canonical.canonicalize` for signing input. See thermocline/.planning/research/PITFALLS.md (Pitfall 11).", file=sys.stderr)
        return 1
    print("ok: no json.dumps found in library code outside allowlist.", file=sys.stderr)
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
