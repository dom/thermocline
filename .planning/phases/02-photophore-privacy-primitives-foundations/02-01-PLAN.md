```python
"""Shared types used across photophore.{audit, channels, classifier, shadow, policy, cli}."""
from __future__ import annotations
from enum import Enum
from typing import NewType

# Tier enum — Plan 02-02 will import from here.
class Tier(Enum):
    LOCAL = "local"
    SHARED = "shared"
    PUBLIC = "public"

# ID NewTypes — opaque strings for type discipline (no runtime overhead).
ChannelId = NewType("ChannelId", str)
```

```python
    BEFORE DELETE ON entries
BEGIN
    SELECT RAISE(ABORT, 'append-only: entries table is immutable by design');
END;
CREATE TRIGGER IF NOT EXISTS entries_no_update
    BEFORE UPDATE ON entries
BEGIN
    SELECT RAISE(ABORT, 'append-only: entries table is immutable by design');
END;
"""

    conn.execute("PRAGMA foreign_keys=ON;")
```

```python
"""BLAKE3 chain hash with versioned algo dispatch (AUDIT-02). Hash domain D-03:
entry_hash = blake3(canonicalize(entry_dict minus entry_hash field)).
"""
from __future__ import annotations
from typing import Callable, Mapping, Any
import blake3
from thermocline import canonicalize
from ..errors import UnsupportedChainAlgoError

ALGO_VERSION_DEFAULT = "blake3-v1"

# Forward-compatible registry. Plan 02-02 / Phase 4 may add "blake3-v2" if redaction-domain semantics change.
_HASH_ALGO_REGISTRY: dict[str, Callable[[bytes], str]] = {
    "blake3-v1": lambda data: blake3.blake3(data).hexdigest(),
}

def compute_hash_by_version(algo_version: str, data: bytes) -> str:
    fn = _HASH_ALGO_REGISTRY.get(algo_version)
    if fn is None:
        raise UnsupportedChainAlgoError(
            f"unknown algo_version: {algo_version!r}; known: {sorted(_HASH_ALGO_REGISTRY)}",
            code="UNSUPPORTED_CHAIN_ALGO",
        )
    return fn(data)

def compute_entry_hash(entry_minus_hash: Mapping[str, Any]) -> str:
    """D-03: hash domain = canonicalize(entry minus entry_hash field).

    Caller is responsible for stripping the entry_hash key before calling.
    """
    if "entry_hash" in entry_minus_hash:
        raise ValueError("compute_entry_hash refuses input that still carries entry_hash field")
    canonical = canonicalize(dict(entry_minus_hash))
    algo = entry_minus_hash.get("algo_version", ALGO_VERSION_DEFAULT)
    return compute_hash_by_version(str(algo), canonical)

def verify_entry_hash(entry: Mapping[str, Any]) -> bool:
    if "entry_hash" not in entry:
        return False
    minus = {k: v for k, v in entry.items() if k != "entry_hash"}
    return compute_entry_hash(minus) == entry["entry_hash"]
```

```python
"""AuditLog: append/query/export/verify with the chain hash domain from _chain.py."""
from __future__ import annotations
import json
import sqlite3
import uuid
from datetime import datetime, timezone
from pathlib import Path
from typing import Iterator, Any
from ..core import KNOWN_EVENT_TYPES
from ..errors import AuditWriteError, AuditChainBrokenError
from ._anchor import AnchorTarget, NullAnchor
from ._chain import compute_entry_hash, verify_entry_hash, ALGO_VERSION_DEFAULT
from ._schema import connect, init
from ._types import AuditEntry, from_dict

class AuditLog:
    def __init__(self, path: Path | str, *, anchor: AnchorTarget | None = None) -> None:
        self._path = str(path)
        self._conn = connect(self._path)
        init(self._conn)
        self._anchor: AnchorTarget = anchor if anchor is not None else NullAnchor()

    @property
    def path(self) -> str:
        return self._path

    def _last_entry_hash(self) -> str:
        cur = self._conn.execute(
            "SELECT entry_hash FROM entries ORDER BY timestamp DESC, id DESC LIMIT 1"
        )
        row = cur.fetchone()
        return row[0] if row else ""  # chain head sentinel = ""

    def append(self, *, event_type: str, channel_id: str | None = None,
               envelope_id: str | None = None, payload: dict[str, Any] | None = None,
               timestamp: str | None = None) -> AuditEntry:
        if event_type not in KNOWN_EVENT_TYPES:
            # Defensive: unknown event_type is a programmer error, not a runtime config error.
            raise AuditWriteError(
                f"unknown event_type {event_type!r}", code="AUDIT_WRITE_FAILED"
            )
        ts = timestamp or datetime.now(tz=timezone.utc).isoformat(timespec="milliseconds").replace("+00:00", "Z")
        prev_hash = self._last_entry_hash()
        entry_id = str(uuid.uuid4())
        entry_minus_hash = {
            "id": entry_id,
            "algo_version": ALGO_VERSION_DEFAULT,
            "prev_hash": prev_hash,
            "event_type": event_type,
            "channel_id": channel_id,
            "envelope_id": envelope_id,
            "timestamp": ts,
            "payload": payload or {},
        }
        entry_hash = compute_entry_hash(entry_minus_hash)
        entry = AuditEntry(entry_hash=entry_hash, **entry_minus_hash)
        try:
            self._conn.execute(
                "INSERT INTO entries (id, algo_version, prev_hash, entry_hash, "
                "event_type, channel_id, envelope_id, timestamp, payload) "
                "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
                entry.to_row(),
            )
        except sqlite3.IntegrityError as exc:
            raise AuditWriteError(
                f"audit log rejected write: {exc}", code="AUDIT_WRITE_FAILED"
            ) from exc
        # AUDIT-07: anchor side-effect (no-op default returns None)
        self._anchor.anchor(entry)
        return entry

    def _query_rows(self, *, channel_id: str | None = None, envelope_id: str | None = None,
                    since: str | None = None, until: str | None = None,
                    event_type: str | None = None, shadow_id: str | None = None,
                    tier: str | None = None, receipt_status: str | None = None,
                    ) -> Iterator[dict[str, Any]]:
        # Build WHERE clause from indexed columns + JSON1 expressions for shadow_id/tier/receipt_status.
        # See 02-RESEARCH.md "JSON1 Query Patterns for AUDIT-05 Filters" for verified SQL.
        clauses: list[str] = []
        params: list[Any] = []
        if channel_id is not None:
            clauses.append("channel_id = ?"); params.append(channel_id)
        if envelope_id is not None:
            clauses.append("envelope_id = ?"); params.append(envelope_id)
        if event_type is not None:
            clauses.append("event_type = ?"); params.append(event_type)
        if since is not None:
            clauses.append("timestamp >= ?"); params.append(since)
        if until is not None:
            clauses.append("timestamp <= ?"); params.append(until)
        if shadow_id is not None:
            clauses.append("EXISTS (SELECT 1 FROM json_each(json_extract(payload, '$.shadow_ids')) WHERE value = ?)")
            params.append(shadow_id)
        if tier is not None:
            clauses.append("EXISTS (SELECT 1 FROM json_each(json_extract(payload, '$.tiers')) WHERE value = ?)")
            params.append(tier)
        if receipt_status is not None:
            clauses.append("json_extract(payload, '$.receipt_status') = ?"); params.append(receipt_status)
        sql = "SELECT id, algo_version, prev_hash, entry_hash, event_type, channel_id, envelope_id, timestamp, payload FROM entries"
        if clauses:
            sql += " WHERE " + " AND ".join(clauses)
        sql += " ORDER BY timestamp ASC, id ASC"
        for row in self._conn.execute(sql, params):
            yield {
                "id": row[0], "algo_version": row[1], "prev_hash": row[2],
                "entry_hash": row[3], "event_type": row[4], "channel_id": row[5],
                "envelope_id": row[6], "timestamp": row[7], "payload": json.loads(row[8]),
            }

    def query(self, **filters: Any) -> list[AuditEntry]:
        # AUDIT-08: verify chain integrity over the returned slice on read.
        rows = list(self._query_rows(**filters))
        for row in rows:
            if not verify_entry_hash(row):
                raise AuditChainBrokenError(
                    f"audit chain broken at entry {row['id']!r}", code="AUDIT_CHAIN_BROKEN"
                )
        # Note: prev_hash sequence check happens in verify_chain() (full pass), not query() (slice).
        # query() detects per-entry tamper via verify_entry_hash; full chain verification = verify_chain().
        return [from_dict(r) for r in rows]

    def verify_chain(self) -> tuple[bool, str | None]:
        """Walk the entire chain. Return (True, head_hash) on success or (False, broken_at_id) on failure."""
        prev = ""
        last_hash = ""
        broken_at: str | None = None
        for row in self._query_rows():
            if row["prev_hash"] != prev:
                broken_at = row["id"]; break
            if not verify_entry_hash(row):
                broken_at = row["id"]; break
            prev = row["entry_hash"]; last_hash = row["entry_hash"]
        if broken_at is not None:
            return False, broken_at
        return True, last_hash

    def export(self) -> Iterator[dict[str, Any]]:
        """JSON Lines export — one dict per line. AUDIT-06 contract."""
        yield from self._query_rows()
```

```python
"""Channel record types and state machine. CHAN-01, CHAN-02."""
from __future__ import annotations
from dataclasses import dataclass, field, replace
from ..core import ChannelId, ChannelState
from ..errors import ChannelStateError

# Valid transitions matrix per CHAN-02 (PROPOSED -> OPEN -> SUSPENDED -> CLOSED).
_VALID_TRANSITIONS: dict[ChannelState, frozenset[ChannelState]] = {
    ChannelState.PROPOSED:  frozenset({ChannelState.OPEN, ChannelState.CLOSED}),  # never opened, can close pre-open
    ChannelState.OPEN:      frozenset({ChannelState.SUSPENDED, ChannelState.CLOSED}),
    ChannelState.SUSPENDED: frozenset({ChannelState.OPEN, ChannelState.CLOSED}),  # resume back to OPEN, or close
    ChannelState.CLOSED:    frozenset(),  # terminal
}

@dataclass(frozen=True)
class Channel:
    id: ChannelId
    local_node: str
    remote_node: str
    ceiling: str            # "tier-0" | "tier-1" | "tier-2"
    key_scheme: str         # "brine" | "pgp" | "x509" | "none"
    state: ChannelState
    created_at: str         # ISO 8601 UTC
    creator_identity: str
    description: str = ""
    remote_pubkey_hex: str | None = None  # public key — safe to store; see Pitfall 9 + Pitfall 4 notes

    def transition_to(self, new_state: ChannelState) -> "Channel":
        allowed = _VALID_TRANSITIONS[self.state]
        if new_state not in allowed:
            raise ChannelStateError(
                f"invalid transition {self.state.value} -> {new_state.value}; "
                f"allowed from {self.state.value}: {sorted(s.value for s in allowed)}",
                code="CHANNEL_STATE_INVALID",
            )
        return replace(self, state=new_state)
```

**`src/photophore/channels/_index.py`** (OQ-1 RESOLVED YES — `_index` sentinel implemented):
```python
"""_index sentinel. python-keyring has no enumerate-by-service API; we maintain
a JSON array at service=photophore.channel, username=_index storing the list of
channel_ids. Updated atomically with channel create/delete.

This is the ONLY way D-05 keystore-as-truth bootstrap can detect channels in the
keystore that are missing from channels.db — without _index, we can only walk
channels.db and verify each entry exists in the keystore, but we cannot detect
keystore entries that have no channels.db row.
"""
from __future__ import annotations
import json
from typing import Sequence
import keyring

_KEYSTORE_SERVICE = "photophore.channel"  # MUST match _keystore.py
_INDEX_SENTINEL_KEY = "_index"

def get_index() -> list[str]:
    raw = keyring.get_password(_KEYSTORE_SERVICE, _INDEX_SENTINEL_KEY)
    if raw is None or raw == "":
        return []
    return list(json.loads(raw))

def set_index(channel_ids: Sequence[str]) -> None:
    keyring.set_password(_KEYSTORE_SERVICE, _INDEX_SENTINEL_KEY, json.dumps(list(channel_ids)))

def add_to_index(channel_id: str) -> None:
    ids = get_index()
    if channel_id not in ids:
        ids.append(channel_id)
        set_index(ids)

def remove_from_index(channel_id: str) -> None:
    ids = [i for i in get_index() if i != channel_id]
    set_index(ids)
```

`ChannelStore` API:
```python
class ChannelStore:
    def __init__(self, channels_db_path: Path | str, audit_log: AuditLog) -> None:
        # channels.db, audit_log, keystore (the three stores per D-04)
        self._channels_db_path = str(channels_db_path)
        self._conn = self._connect_and_init(self._channels_db_path)
        self._audit = audit_log
        # Refuse to start if keystore unavailable (Phase 1 BL-03 isinstance probe)
        from ._keystore import _probe_keystore
        _probe_keystore()

    @property
    def channels_db_path(self) -> str:
        return self._channels_db_path

    def create(self, *, remote_node: str, ceiling: str, key_scheme: str,
               local_node: str, creator_identity: str, description: str = "",
               remote_pubkey_hex: str | None = None) -> Channel:
        from . import _keystore as ks
        from . import _index as idx
        channel_id = ChannelId(str(uuid.uuid4()))
        now = datetime.now(tz=timezone.utc).isoformat(timespec="milliseconds").replace("+00:00", "Z")
        channel = Channel(
            id=channel_id, local_node=local_node, remote_node=remote_node,
            ceiling=ceiling, key_scheme=key_scheme, state=ChannelState.PROPOSED,
            created_at=now, creator_identity=creator_identity,
            description=description, remote_pubkey_hex=remote_pubkey_hex,
        )
        # D-07 STEP 1: keystore set (authoritative trust store)
        ks._set_channel(channel_id, _channel_to_dict(channel))
        idx.add_to_index(channel_id)  # _index sentinel update — same step (OQ-1)
        # D-07 STEP 2: audit append (BEFORE step 3 — CHAN-05 mandates audit before report)
        self._audit.append(
            event_type=AuditEventType.CHANNEL_CREATED,
            channel_id=channel_id,
            payload={
                "remote_node": remote_node, "ceiling": ceiling, "key_scheme": key_scheme,
                "local_node": local_node, "creator_identity": creator_identity,
                "created_at": now, "description": description,
                "remote_pubkey_hex": remote_pubkey_hex,
            },
        )
        # D-07 STEP 3: channels.db upsert (derived projection)
        self._upsert_channels_db(channel)
        return channel

    def list_channels(self) -> list[Channel]:
        # W9: explicit implementation (was `...` placeholder).
        # Read from channels.db (the index) for the row set, then resolve each row's
        # full canonical record from the keystore (authoritative per D-05 keystore-as-truth).
        cur = self._conn.execute(
            "SELECT id, local_node, remote_node, ceiling, key_scheme, state, created_at, updated_at "
            "FROM channels ORDER BY created_at"
        )
        rows = cur.fetchall()
        out: list[Channel] = []
        from . import _keystore as ks
        for row in rows:
            full_record = ks._get_channel(row[0])
            if full_record is None:
                # channels.db row references a missing keystore entry — drift that bootstrap()
                # would have caught at startup; here we surface a hard error so the operator sees it.
                raise KeystoreError(
                    f"channels.db row references missing keystore entry: {row[0]!r}",
                    code="CHANNEL_NOT_FOUND",
                )
            out.append(_dict_to_channel(full_record))
        return out

    def show(self, channel_id: ChannelId) -> Channel:
        # Pull from KEYSTORE (authoritative) — show is a deep read, walking through the trust store.
        from . import _keystore as ks
        record = ks._get_channel(channel_id)
        if record is None:
            raise KeystoreError(f"channel {channel_id!r} not found in keystore", code="CHANNEL_NOT_FOUND")
        return _dict_to_channel(record)

    def transition_to(self, channel_id: ChannelId, new_state: ChannelState) -> Channel:
        old = self.show(channel_id)
        new = old.transition_to(new_state)  # raises ChannelStateError if invalid (CHAN-02)
        # D-07 three-step ordering applies here too:
        from . import _keystore as ks
        ks._set_channel(channel_id, _channel_to_dict(new))
        # No _index update on transition — id stays in index until full delete (CHAN-02 "IDs never reused")
        event = {
            ChannelState.OPEN:      AuditEventType.CHANNEL_OPENED,
            ChannelState.SUSPENDED: AuditEventType.CHANNEL_SUSPENDED,
            ChannelState.CLOSED:    AuditEventType.CHANNEL_CLOSED,
            ChannelState.PROPOSED:  AuditEventType.CHANNEL_CREATED,  # never reached at this point
        }[new_state]
        self._audit.append(event_type=event, channel_id=channel_id, payload={"from_state": old.state.value, "to_state": new.state.value})
        self._upsert_channels_db(new)
        return new

    def set_ceiling(self, channel_id: ChannelId, new_ceiling: str) -> Channel:
        # CHAN-03: distinct event types for raise vs lower.
        old = self.show(channel_id)
        old_idx = _CEILING_RANK[old.ceiling]
        new_idx = _CEILING_RANK[new_ceiling]
        if new_idx == old_idx:
            return old  # no-op; do NOT emit an audit event
        new = replace(old, ceiling=new_ceiling)
        from . import _keystore as ks
        ks._set_channel(channel_id, _channel_to_dict(new))
        event = (AuditEventType.CHANNEL_CEILING_RAISED
                 if new_idx > old_idx else AuditEventType.CHANNEL_CEILING_LOWERED)
        self._audit.append(event_type=event, channel_id=channel_id,
                           payload={"from_ceiling": old.ceiling, "to_ceiling": new.ceiling})
        self._upsert_channels_db(new)
        return new
```

```python
"""bootstrap(): walks the _index sentinel, verifies each channel has a corresponding
channel.created audit entry, rebuilds channels.db from keystore on drift detection.

Halt-on-mismatch: if a channel exists in the keystore but no channel.created event
in the audit log, raise UnauditedChannelError. The node refuses to operate until
manual reconciliation.
"""
from __future__ import annotations
from . import _index as idx
from . import _keystore as ks
from ..audit import AuditLog
from ..core import AuditEventType
from ..errors import UnauditedChannelError

def bootstrap(audit_log: AuditLog, channel_store_db_path: str) -> dict:
    """Returns {bootstrapped: <count>, rebuilt: <count>}. Raises UnauditedChannelError on drift."""
    keystore_ids = idx.get_index()
    rebuilt = 0
    for channel_id in keystore_ids:
        # Verify keystore record exists for this id (defensive — _index could be stale)
        record = ks._get_channel(channel_id)
        if record is None:
            # Index has an orphan — strip from index, but DO NOT halt (this is index drift, not unaudited channel)
            idx.remove_from_index(channel_id)
            continue
        # Verify a channel.created audit entry exists for this channel_id
        events = audit_log.query(channel_id=channel_id, event_type=AuditEventType.CHANNEL_CREATED)
        if not events:
            raise UnauditedChannelError(
                f"channel {channel_id!r} exists in keystore but has no channel.created audit entry; "
                f"halt for manual reconciliation",
                code="UNAUDITED_CHANNEL",
            )
        # Rebuild channels.db row from keystore record
        # ... INSERT OR REPLACE into channels table
        rebuilt += 1
    return {"bootstrapped": len(keystore_ids), "rebuilt": rebuilt}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Library API to caller | audit.append(payload=...) accepts arbitrary dicts; canonicalize at hash time, not at API boundary. |
| Audit log file to filesystem | SQLite file at audit.db — readable by anyone with filesystem access; integrity proven by chain hash, not access control. |
| Trust store (keystore) to filesystem | Platform keystore (Keychain / libsecret / Credential Manager) — separate backing from audit.db AND channels.db (D-04). |
| Channel index (channels.db) to keystore | channels.db is a derived projection; keystore is authoritative. Drift detection rebuilds channels.db from keystore. |
| _index sentinel to enumeration | python-keyring has no enumerate-by-service; _index is the single source of truth for "which channel_ids exist in the keystore." |
| Audit log to verifier | verify_chain() walks the chain; any byte mutation invalidates the entry's hash, which mismatches the next entry's prev_hash. |
