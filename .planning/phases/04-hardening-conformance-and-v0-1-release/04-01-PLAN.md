```python
#!/usr/bin/env python3
"""AT-C* coverage gate for thermocline (AT-C1..C6).

Globs thermocline/thermocline/python/tests/at_negative/test_at_c*.py and asserts
all six AT-C surfaces have at least one test file.
"""
from __future__ import annotations
import re, sys
from pathlib import Path

EXPECTED: frozenset[str] = frozenset({"AT-C1","AT-C2","AT-C3","AT-C4","AT-C5","AT-C6"})
PATTERN = re.compile(r"^test_at_c(\d+)_")
ROOT = Path(__file__).resolve().parents[1] / "thermocline" / "python" / "tests" / "at_negative"

def main() -> int:
    if not ROOT.is_dir():
        print(f"FAIL: {ROOT} does not exist", file=sys.stderr)
        return 1
    found: set[str] = set()
    for p in sorted(ROOT.glob("test_at_*.py")):
        m = PATTERN.match(p.name.lower())
        if m:
            found.add(f"AT-C{m.group(1)}")
    missing = EXPECTED - found
    if missing:
        print(f"FAIL: missing AT-C coverage: {sorted(missing)}", file=sys.stderr)
        return 1
    print(f"ok: AT-C coverage complete ({len(found)}/6).", file=sys.stderr)
    return 0

if __name__ == "__main__":
    sys.exit(main())
```

```python
#!/usr/bin/env python3
"""Cross-repo AT-* coverage roll-up (17/17 union).

Reads THERMOCLINE_SUITE_ROOT (default $HOME/Projects/dom). Subprocess-invokes each
repo's tools/at_coverage.py. Exits 0 only when all three pass.

def main() -> int:
    failed = []
    for repo, tool in REPOS:
        path = SUITE_ROOT / repo / tool
        if not path.is_file():

        return 1
```

```python
        self.generic_visit(node)
```

```python
"""CLI-06 (D-07): helper for emitting cli.invoked audit entries."""
from __future__ import annotations
from typing import Any
from photophore.core import AuditEventType
from ._store import AuditLog

    outcome: str,
    exit_code: int,
    ts: str,

    """Record a CLI subcommand invocation (CLI-06 / D-07).

        envelope_id=None,
        payload={

            "args": args,
            "outcome": outcome,
            "exit_code": exit_code,
        },
        timestamp=ts,
    )
```
